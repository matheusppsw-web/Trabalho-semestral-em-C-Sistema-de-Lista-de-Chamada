#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define TAM_NOME 100

typedef struct {
    int matricula;
    char nome[TAM_NOME];
    float presenca;
} Aluno;

void cadastrarAlunos(Aluno **lista, int *quantidade);
void listarAlunos(Aluno *lista, int quantidade);
void buscarAluno(Aluno *lista, int quantidade);
void adicionarNovosAlunos(Aluno **lista, int *quantidade);

int main() {
    Aluno *lista = NULL;
    int quantidade = 0;
    int opcao;

    do {
        printf("\n====== SISTEMA DE LISTA DE CHAMADA ======\n");
        printf("1. Cadastrar alunos\n");
        printf("2. Listar alunos\n");
        printf("3. Buscar aluno pelo nome\n");
        printf("4. Adicionar novos alunos\n");
        printf("5. Sair\n");
        printf("Escolha uma opcao: ");
        scanf("%d", &opcao);
        getchar();

        switch (opcao) {
            case 1:
                cadastrarAlunos(&lista, &quantidade);
                break;
            case 2:
                listarAlunos(lista, quantidade);
                break;
            case 3:
                buscarAluno(lista, quantidade);
                break;
            case 4:
                adicionarNovosAlunos(&lista, &quantidade);
                break;
            case 5:
                printf("Encerrando o programa...\n");
                break;
            default:
                printf("Opcao invalida!\n");
        }

    } while (opcao != 5);

    free(lista);
    return 0;
}

void cadastrarAlunos(Aluno **lista, int *quantidade) {
    printf("Quantos alunos deseja cadastrar inicialmente? ");
    scanf("%d", quantidade);
    getchar();

    *lista = (Aluno*) malloc((*quantidade) * sizeof(Aluno));

    if (*lista == NULL) {
        printf("Erro na alocacao de memoria!\n");
        exit(1);
    }

    for (int i = 0; i < *quantidade; i++) {
        printf("\nAluno %d:\n", i + 1);
        printf("Matricula: ");
        scanf("%d", &(*lista)[i].matricula);
        getchar();
        printf("Nome: ");
        fgets((*lista)[i].nome, TAM_NOME, stdin);
        (*lista)[i].nome[strcspn((*lista)[i].nome, "\n")] = '\0';
        printf("Porcentagem de presenca: ");
        scanf("%f", &(*lista)[i].presenca);
        getchar();
    }
}

void listarAlunos(Aluno *lista, int quantidade) {
    if (quantidade == 0) {
        printf("Nenhum aluno cadastrado.\n");
        return;
    }

    printf("\n==== LISTA DE ALUNOS ====\n");
    for (int i = 0; i < quantidade; i++) {
        printf("%d - %s (%.1f%% presenca)\n",
               lista[i].matricula, lista[i].nome, lista[i].presenca);
    }
}

void buscarAluno(Aluno *lista, int quantidade) {
    if (quantidade == 0) {
        printf("Nenhum aluno cadastrado.\n");
        return;
    }

    char nomeBusca[TAM_NOME];
    printf("Digite o nome do aluno que deseja buscar: ");
    fgets(nomeBusca, TAM_NOME, stdin);
    nomeBusca[strcspn(nomeBusca, "\n")] = '\0';

    int encontrado = 0;

    for (int i = 0; i < quantidade; i++) {
        if (strcmp(lista[i].nome, nomeBusca) == 0) {
            printf("\nAluno encontrado:\n");
            printf("Matricula: %d\n", lista[i].matricula);
            printf("Nome: %s\n", lista[i].nome);
            printf("Presenca: %.1f%%\n", lista[i].presenca);
            encontrado = 1;
            break;
        }
    }

    if (!encontrado)
        printf("Aluno nao encontrado.\n");
}

void adicionarNovosAlunos(Aluno **lista, int *quantidade) {
    int novos;
    printf("Quantos alunos deseja adicionar? ");
    scanf("%d", &novos);
    getchar();

    *lista = (Aluno*) realloc(*lista, (*quantidade + novos) * sizeof(Aluno));

    if (*lista == NULL) {
        printf("Erro ao realocar memoria!\n");
        exit(1);
    }

    for (int i = *quantidade; i < *quantidade + novos; i++) {
        printf("\nNovo aluno %d:\n", i + 1);
        printf("Matricula: ");
        scanf("%d", &(*lista)[i].matricula);
        getchar();
        printf("Nome: ");
        fgets((*lista)[i].nome, TAM_NOME, stdin);
        (*lista)[i].nome[strcspn((*lista)[i].nome, "\n")] = '\0';
        printf("Porcentagem de presenca: ");
        scanf("%f", &(*lista)[i].presenca);
        getchar();
    }

    *quantidade += novos;
}
